                            @Override
                            public DefaultResult editCard(EditCardRequest request) {
                                DefaultResult result = new DefaultResult();
                                if(!loginService.checkUserExists(request.getUserName())) {
                                    result.setErrorCode(1);
                                    result.setErrorMessage("User does not exist!");
                                    return result;
                                }
                                
                                CardWithOwner card = cardService.getCard(request.getCardId());
                                if(!card.getOwnerName().equals(request.getUserName())) {
                                    result.setErrorCode(2);
                                    result.setErrorMessage("User " + request.getUserName() + " is not the owner of card " + request.getCardId());
                                    return result;
                                }
                                
                                if(!cardService.editCard(request)) {
                                    result.setErrorCode(3);
                                    result.setErrorMessage("Error editing card " + request.getCardId());
                                    return result;
                                }
                                
                                return result;
                            }
